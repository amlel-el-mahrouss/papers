\begin{equation}
    A=\sum_{n}^{p} ((n \cdot p)!) < l

\begin{equation}
    A = p! + \text{ } (n+1p)! + \text{ } (n+2p)! + \text{ } \dots \text{ }+ (n+lp)! 
\end{equation}
